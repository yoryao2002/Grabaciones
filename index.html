// ============================================================ // CONFIGURACIÓN // ============================================================ function getConfig() { const properties = PropertiesService.getScriptProperties(); return { ASSEMBLYAI_API_KEY: properties.getProperty('ASSEMBLYAI_API_KEY') || '', SUPPORTED_EXTENSIONS: ['.mp3', '.wav', '.m4a', '.flac', '.ogg', '.webm', '.mp4', '.aac', '.wma'], LANGUAGE: 'es' }; } // ============================================================ // RECEPTOR DE AUDIO (PARA EL GRABADOR) // ============================================================ function doPost(e) { try { const params = JSON.parse(e.postData.contents); const { nombre, audioBase64, tipo } = params; if (!nombre || !audioBase64) { return ContentService .createTextOutput(JSON.stringify({ success: false, message: 'Faltan datos' })) .setMimeType(ContentService.MimeType.JSON); } // 📁 CARPETA DONDE SE GUARDARÁN LOS AUDIOS const folder = DriveApp.getFolderById('1lMMVyVI6vzUHw-KHGDY4gJW0u7Ud-GLV'); // Crear nombre único const fecha = new Date(); const idUnico = 'A' + fecha.getTime().toString(36) + Math.random().toString(36).substr(2, 4); const extension = tipo || 'wav'; const nombreArchivo = `${nombre}_${fecha.getTime()}_${idUnico}.${extension}`; // Decodificar y guardar const audioBlob = Utilities.newBlob(Utilities.base64Decode(audioBase64), `audio/${extension}`, nombreArchivo); const file = folder.createFile(audioBlob); // Guardar en Sheets try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); sheet.appendRow([ fecha.toLocaleString('es-ES', { timeZone: 'America/Tegucigalpa' }), nombre, extension.toUpperCase(), file.getUrl(), idUnico ]); } catch(e) { console.log('Error en Sheets:', e); } return ContentService .createTextOutput(JSON.stringify({ success: true, message: 'Audio guardado', url: file.getUrl() })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService .createTextOutput(JSON.stringify({ success: false, message: 'Error: ' + error.message })) .setMimeType(ContentService.MimeType.JSON); } } // ============================================================ // FUNCIÓN PARA PROBAR (GET) // ============================================================ function doGet() { return ContentService .createTextOutput('✅ Servidor activo. Envía datos con POST.') .setMimeType(ContentService.MimeType.TEXT); }
